def combineDF(df1, df2, df3, df4):
    df1 = df1.union(df2)
    df1 = df1.union(df3)
    df1 = df1.union(df4)
    return df1

bikeDF = combineDF(bike_Q1, bike_Q2, bike_Q3, bike_Q4)

    • Midnight: 0 - 3 hours
    • Early Morning: 4 - 7 hours
    • Morning: 8 - 11 hours
    • Noon: 12 - 15 hours
    • Evening: 16 - 19 hours
    • Night: 20 - 23 hours

    a. Display the total count of races that occurred during the time_of_day over the entire year

    b. Every quarter, display the hour at which most of the races took place. Mention the count

timestamp_time_of_day = timestamp_data.withColumn("time_of_day",\
    f.when((f.col("start_hour") == 0) | (f.col("start_hour") == 1) | \
           (f.col("start_hour") == 2) | (f.col("start_hour") == 3),"midnight: 0-3")\
    .when((f.col("start_hour") == 4) | (f.col("start_hour") == 5) | \
          (f.col("start_hour") == 6) | (f.col("start_hour") == 7),"early morning: 4-7")\
    .when((f.col("start_hour") == 8) | (f.col("start_hour") == 9) | \
          (f.col("start_hour") == 10) | (f.col("start_hour") == 11),"morning: 8-11")\
    .when((f.col("start_hour") == 12) | (f.col("start_hour") == 13) | \
          (f.col("start_hour") == 14) | (f.col("start_hour") == 15),"noon: 12-15")\
    .when((f.col("start_hour") == 16) | (f.col("start_hour") == 17) | \
          (f.col("start_hour") == 18) | (f.col("start_hour") == 19),"evening: 16-19")\
    .when((f.col("start_hour") == 20) | (f.col("start_hour") == 21) | \
          (f.col("start_hour") == 22) | (f.col("start_hour") == 23),"night: 20-13")\
    .otherwise("NA"))
