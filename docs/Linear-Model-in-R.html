The work consists of analyzing a big dataset (72 csv files) with hourly pollution data from Madrid, from period 2011 to 2016. This dataset consists of several pollutants, whose measures are taken by different stations every hour.

The weather dataset is a time series consisting of minimum, maximum and average values of temperature, precipitation,humidity and wind speeds in Madrid.

Read_Madrid_Data <- function(y, m) {
  ## Obtain filename of files for the year & month
  Myfiles <- paste0('hourly_data_', y, '_', m, '.csv')
## Load for this year/month
  z <- read.csv(Myfiles)
  z$date <- as.Date(paste0('20', y, '-', m, '-', z$day))
  z$day <- NULL
  return(z)
}
Entire_Madrid_Data <- function(ys,ms) {
  pollutants_H <- data.frame();
  for(y in ys) {
    for(m in ms) {

      monthly <- Read_Madrid_Data(y,m)
#Join
      pollutants_H <- rbind(pollutants_H, monthly )
    }
  }
  return(pollutants_H)
}
